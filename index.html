// 参考代码示例：两数交换
#include <stdio.h>
int main() {
    int a, b, t;
    scanf("%d%d", &a, &b);
    t = a; a = b; b = t;
    printf("%d %d", a, b);
    return 0;
}

解释：利用临时变量t，先把a的值存起来，再把b赋值给a，最后把t赋值给b，完成交换。

// 例题1.2：求两数最大者（单分支+双分支两种写法）
#include <stdio.h>
int main() {
    int a, b, max;
    scanf("%d%d", &a, &b);

    // 写法1：单分支if语句
    max = a;
    if (b > max) {
        max = b;
    }

    // 写法2：双分支if-else语句
    /*
    if (a > b) {
        max = a;
    } else {
        max = b;
    }
    */

    printf("%d", max);
    return 0;
}

解释：单分支写法先假设a是最大值，再判断b是否更大；双分支写法直接比较a和b，把大的赋值给max。两种方法都能得到两数的最大值。

// 例题1.5：数列求和（循环结构）
#include <stdio.h>
int main() {
    int n, i, sum = 0;
    scanf("%d", &n);
    for(i = 1; i <= n; i++) {
        sum += i;
    }
    printf("%d", sum);
    return 0;
}

解释：用for循环从1加到n，每次把i的值累加到sum里，最后输出总和。也可以用公式n*(n+1)/2直接计算。

    long long n, res = 0;
    int sign = 1;
    scanf("%lld", &n);

        sign = -1;
        n = -n;

}

解释：先记录正负号，把负数转为正数处理；用循环每次取个位数字，拼接到新数上；最后还原正负号，自动去掉前导零。

// 实践1.1：时间管理大师
#include <stdio.h>
int main() {
    int a, b, c, d;
    int start, end, diff;
    scanf("%d%d%d%d", &a, &b, &c, &d);

    // 换算成分钟
    start = a * 60 + b;
    end = c * 60 + d;
    diff = end - start;

    int e = diff / 60;
    int f = diff % 60;
    printf("%d %d", e, f);
    return 0;
}

解释：先把开始时间和结束时间都换算成分钟，相减得到总分钟差，再换算回小时和分钟，输出结果。

// 实践1.3：ISBN号码（NOIP2018普及组）
#include <stdio.h>
#include <string.h>
int main() {
    char isbn[15];
    scanf("%s", isbn);
    int sum = 0, k = 1;
    // 计算前9位数字的加权和
    for(int i = 0; i < strlen(isbn) - 1; i++) {
        if(isbn[i] != '-') {
            sum += (isbn[i] - '0') * k++;
        }
    }
    int mod = sum % 11;
    char correct;
    if(mod == 10) correct = 'X';
    else correct = mod + '0';
    // 判断识别码是否正确
    if(isbn[strlen(isbn)-1] == correct) {
        printf("Right");
    } else {
        isbn[strlen(isbn)-1] = correct;
        printf("%s", isbn);
    }
    return 0;
}

解释：先遍历ISBN字符串，跳过分隔符，计算前9位数字的加权和；再用和除以11求余，得到正确识别码；最后对比输入的识别码，输出Right或修正后的ISBN。

// 实践1.2：买铅笔（NOIP2016普及组）
#include <stdio.h>
int main() {
    int n, min_cost = 1000000000;
    int num[3], price[3];
    scanf("%d", &n);
    for(int i = 0; i < 3; i++) {
        scanf("%d%d", &num[i], &price[i]);
    }
    for(int i = 0; i < 3; i++) {
        int need = (n + num[i] - 1) / num[i]; // 向上取整
        int cost = need * price[i];
        if(cost < min_cost) {
            min_cost = cost;
        }
    }
    printf("%d", min_cost);
    return 0;
}

解释：分别计算三种包装的购买费用，用向上取整算出需要购买的包数，再乘以单价得到总费用，最后取三种费用的最小值。

// 例题2.4：珠心算测试（NOIP2014普及组）
#include <stdio.h>
#include <stdbool.h>
int main() {
    int n, i, j, k, count = 0;
    int a[105];
    bool visit[10005] = {false}; // 标记是否已统计过

    scanf("%d", &n);
    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // 枚举两个不同的数相加
    for(i = 0; i < n; i++) {
        for(j = i + 1; j < n; j++) {
            int sum = a[i] + a[j];
            // 检查sum是否在数组中且未被统计过
            for(k = 0; k < n; k++) {
                if(a[k] == sum && !visit[k]) {
                    visit[k] = true;
                    count++;
                }
            }
        }
    }

    printf("%d", count);
    return 0;
}

解释：用三重循环枚举两个不同的数相加，再检查和是否在数组中；用visit数组标记已统计过的数，避免重复计数，最后输出符合条件的总数。

// 实践2.3：校门外的树
#include <stdio.h>
#include <string.h>

int main() {
    int n, m;
    scanf("%d%d", &n, &m);
    int visit[10001]; // 标记树是否被移除
    memset(visit, 0, sizeof(visit));

    for(int i = 0; i < m; i++) {
        int u, v;
        scanf("%d%d", &u, &v);
        for(int j = u; j <= v; j++) {
            visit[j] = 1; // 标记该位置的树要被移除
        }
    }

    int count = 0;
    for(int i = 0; i <= n; i++) {
        if(visit[i] == 0) {
            count++;
        }
    }

    printf("%d", count);
    return 0;
}

解释：用数组visit标记每个位置的树是否被移除，读入建房区间后，将区间内的树全部标记为移除，最后统计未被标记的树的数量，即为剩余树木数量。

// 实践2.4：神奇的幻方（NOIP2015提高组）
#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int magic[40][40] = {0};
    int x, y;
    // 1放在第一行中间
    x = 0;
    y = n / 2;
    magic[x][y] = 1;

    for(int k = 2; k <= n * n; k++) {
        int nx = x - 1;
        int ny = y + 1;
        if(nx < 0 && ny < n) {
            // 情况1：上一行不在，放最后一行，右一列
            nx = n - 1;
        } else if(ny >= n && nx >= 0) {
            // 情况2：右一列不在，放第一列，上一行
            ny = 0;
        } else if(nx < 0 && ny >= n) {
            // 情况3：右上角，放正下方
            nx = x + 1;
            ny = y;
        } else if(magic[nx][ny] != 0) {
            // 情况4：右上方有数，放正下方
            nx = x + 1;
            ny = y;
        }
        magic[nx][ny] = k;
        x = nx;
        y = ny;
    }

    // 输出幻方
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            if(j > 0) printf(" ");
            printf("%d", magic[i][j]);
        }
        printf("\n");
    }
    return 0;
}

解释：用二维数组模拟幻方构造，先把1放在第一行中间，然后按题目给的4条规则依次放置后续数字；每次根据上一个数的位置计算下一个数的坐标，直到填满整个矩阵，最后按格式输出。

// 例题3.1：回文字符串
#include <stdio.h>
#include <string.h>
int main() {
    char s[101];
    int caseNum = 1;
    while(scanf("%s", s) != EOF) {
        int len = strlen(s);
        int isPalindrome = 1;
        for(int i = 0; i < len / 2; i++) {
            if(s[i] != s[len - 1 - i]) {
                isPalindrome = 0;
                break;
            }
        }
        printf("case%d: %s\n", caseNum++, isPalindrome ? "yes" : "no");
    }
    return 0;
}

解释：循环读取每个字符串，通过比较首尾对称位置的字符判断是否为回文串，按题目格式输出case编号和yes/no。

#include <string.h>
int main() {
    int n;

    getchar(); // 吸收换行符
    int count[26] = {0};

            }
        }

    for(int i = 0; i < 26; i++) {
        if(count[i] > max) {
            max = count[i];
        }
    }

    // 输出所有出现次数为max的字母
    for(int i = 0; i < 26; i++) {
        if(count[i] == max) {
            printf("%c %d\n", 'a' + i, max);
        }
    }
    return 0;
}

解释：先读入学生人数，再逐行读取姓名；用数组统计所有小写字母出现次数，找到最大值后，分行输出所有出现次数为最大值的字母和次数。

// 实践3.4：数字字符反转
#include <stdio.h>
#include <string.h>
int main() {
    char s[100];
    scanf("%s", s);
    int len = strlen(s);
    int start = 0;
    // 处理负号
    if(s[0] == '-') {
        printf("-");
        start = 1;
    }
    // 找到末尾第一个非0字符
    int end = len - 1;
    while(end >= start && s[end] == '0') {
        end--;
    }
    // 全0的情况
    if(end < start) {
        printf("0");
    } else {
        // 从end倒序输出
        for(int i = end; i >= start; i--) {
            printf("%c", s[i]);
        }
    }
    return 0;
}

解释：先判断是否为负数，处理负号；再从字符串末尾找到第一个非0字符，从该位置倒序输出数字；若全为0则直接输出0，避免反转后出现前导0。

// 实践3.5：打印字符柱状图
#include <stdio.h>
int main() {
    int count[26] = {0};
    char line[201];
    // 读入5行字符
    for(int i = 0; i < 5; i++) {
        gets(line);
        for(int j = 0; line[j] != '\0'; j++) {
            if(line[j] >= 'A' && line[j] <= 'Z') {
                count[line[j] - 'A']++;
            }
        }
    }
    // 找出现次数最大值
    int max = 0;
    for(int i = 0; i < 26; i++) {
        if(count[i] > max) {
            max = count[i];
        }
    }
    // 从高到低打印柱状图
    for(int i = max; i > 0; i--) {
        for(int j = 0; j < 26; j++) {
            if(count[j] >= i) {
                printf("* ");
            } else {
                printf("  ");
            }
        }
        printf("\n");
    }
    // 打印底部字母
    for(int i = 0; i < 26; i++) {
        printf("%c ", 'A' + i);
    }

        scanf("%s", map[i]);
    }

    int w[55] = {0}, b[55] = {0}, r[55] = {0};
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            if(map[i][j] != 'W') w[i]++;
            if(map[i][j] != 'B') b[i]++;
            if(map[i][j] != 'R') r[i]++;
        }
    }

    // 前缀和优化
    int preW[55] = {0}, preB[55] = {0}, preR[55] = {0};
    for(int i = 0; i < n; i++) {
        preW[i+1] = preW[i] + w[i];
        preB[i+1] = preB[i] + b[i];
        preR[i+1] = preR[i] + r[i];
    }

    int ans = 1e9;

            int cost = preW[i+1] + (preB[j+1] - preB[i+1]) + (preR[n] - preR[j+1]);
            if(cost < ans) ans = cost;
        }
    }

    printf("%d", ans);
    return 0;
}

// 例题7.1：简单选择排序

}

// 例题7.2：冒泡排序
#include <stdio.h>

int main() {
    int n, i, j, temp;
    int a[10005];
    scanf("%d", &n);
    for(i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    // 冒泡排序（升序，带优化）
    for(i = 0; i < n - 1; i++) {
        int swapped = 0;
        for(j = 0; j < n - 1 - i; j++) {
            if(a[j] > a[j + 1]) {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
                swapped = 1;
            }
        }
        if(!swapped) break;
    }

    for(i = 0; i < n; i++) {
        if(i > 0) printf(" ");
        printf("%d", a[i]);
    }
    return 0;
}

解释：采用带优化的冒泡排序，通过多趟遍历未排序部分，比较并交换相邻元素，将较大元素逐步“冒泡”到末尾；若某趟遍历无交换发生，说明序列已有序，可提前结束排序。

// 例题7.4：快速排序
#include <stdio.h>

// 快速排序
void quickSort(int a[], int l, int r) {
    if(l >= r) return;
    int pivot = a[l];
    int i = l, j = r;
    while(i < j) {
        while(i < j && a[j] >= pivot) j--;
        a[i] = a[j];
        while(i < j && a[i] <= pivot) i++;
        a[j] = a[i];
    }
    a[i] = pivot;
    quickSort(a, l, i-1);
    quickSort(a, i+1, r);
}

int main() {
    int n, a[10005];
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    quickSort(a, 0, n-1);
    for(int i = 0; i < n; i++) {
        if(i > 0) printf(" ");
        printf("%d", a[i]);
    }
    return 0;
}

解释：采用分治思想，选第一个数为基准pivot，将数组分为小于基准、大于基准两部分；递归对左右区间排序，最终得到升序序列。

// 例题7.5：二路归并排序
#include <stdio.h>
int temp[10005];

// 合并左右两个有序区间
void merge(int a[], int l, int mid, int r) {
    int i = l, j = mid + 1, k = 0;
    while(i <= mid && j <= r) {
        if(a[i] <= a[j]) temp[k++] = a[i++];
        else temp[k++] = a[j++];
    }
    while(i <= mid) temp[k++] = a[i++];
    while(j <= r) temp[k++] = a[j++];
    for(i = l, k = 0; i <= r; i++, k++)
        a[i] = temp[k];
}

// 归并排序递归拆分
void mergeSort(int a[], int l, int r) {
    if(l >= r) return;
    int mid = (l + r) / 2;
    mergeSort(a, l, mid);
    mergeSort(a, mid+1, r);
    merge(a, l, mid, r);
}

int main() {
    int n, a[10005];
    scanf("%d", &n);
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    mergeSort(a, 0, n-1);
    for(int i = 0; i < n; i++) {
        if(i > 0) printf(" ");
        printf("%d", a[i]);
    }
    return 0;
}

解释：采用分治思想，将数组不断拆分为左右两半，递归排序；再把两个有序区间合并成一个有序区间，最终完成整体升序排序。

// 实践8.6：通用函数的应用
#include <stdio.h>

struct Player {
    char name[21];
    int num;
};

// 输入函数
void input(struct Player *p, int n) {
    for(int i = 0; i < n; i++) {
        scanf("%s %d", (p+i)->name, &(p+i)->num);
    }
}

// 通用排序函数：c=1升序，c=2降序
void sort(struct Player *p, int n, int c) {
    for(int i = 0; i < n-1; i++) {
        for(int j = 0; j < n-1-i; j++) {
            int flag = 0;
            if(c == 1) flag = (p[j].num > p[j+1].num);   // 升序
            else flag = (p[j].num < p[j+1].num);        // 降序
            if(flag) {
                struct Player t = p[j];
                p[j] = p[j+1];
                p[j+1] = t;
            }
        }
    }
}

// 输出函数
void output(struct Player *p, int n) {
    for(int i = 0; i < n; i++) {
        printf("%s %d\n", (p+i)->name, (p+i)->num);
    }
}

int main() {
    int n, c;
    struct Player p[20];
    scanf("%d", &n);
    input(p, n);
    scanf("%d", &c);
    sort(p, n, c);
    output(p, n);
    return 0;
}

解释：按要求拆分输入、通用排序、输出三个函数，用结构体指针传参；排序函数通过参数c控制升降序逻辑，实现一套代码两种排序效果，最后按格式输出球员信息。
